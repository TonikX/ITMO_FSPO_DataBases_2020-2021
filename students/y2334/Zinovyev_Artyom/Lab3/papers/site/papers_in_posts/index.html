    Create table

    CREATE TABLE papers_in_posts (
    id SERIAL,
    post_id INT NOT NULL,
    paper_id INT NOT NULL,
    count INT NOT NULL,
    FOREIGN KEY (post_id) REFERENCES posts(id),
    FOREIGN KEY (paper_id) REFERENCES papers(id)
);
